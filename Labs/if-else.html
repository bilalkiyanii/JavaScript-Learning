Q.3:

Write an if/else statement with the following condition: If the variable age is greater than 18, output "Old enough", otherwise output "Too young"..

A:
var user = +prompt("Please enter your age", "Age");
if (user > 18){
alert("Old enough");
}else{
alert("Too young");
}

Q.4:

Write a program that prompts the user for their name, and then displays a special greeting to that person if their name is the same as yours. If the name entered by the user is anything other than your name, your code should not produce any output.

A:
var name = "bilal"
var userInput = prompt("Please enter your name", "Name");
userInput = userInput.toLowerCase();
if (userInput === name ){
alert("Hi..! Bilal Nice to meet you");
}

Q.7:

Write a program to create a calculator for +, -, *, /, % using switch case statements. (number1, number2 and operator will be input)

A:
var num1 = +prompt("Please enter a number", "1st Number");
var num2 = +prompt("Please enter a number", "2nd Number");
var operator = prompt("Please enter a operator", "Operator");
var result;
switch(operator){
case "+":
result = (num1 + num2);
alert(result);
break;
case "-":
result = (num1 - num2);
alert(result);
break;
case "*":
result = (num1 * num2);
alert(result);
break;
case "/":
result = (num1 / num2);
alert(result);
break;
case "%":
result = (num1 % num2);
alert(result);
break;
default:
alert("Please the following opeartors '+, -, *, /, %' ");
}

Q.8:

Write a program that takes time as input from user in 24 hours clock format like: 1900 = 7pm. Implement the following case using if, else & else if statements

A:
var user = prompt("Please enter a time", "Time");
if (user >= 00 && user < 12){
alert("Good Morning Dear ...!")
}else if (user >=12 && user < 17){
alert("Good Afternoon Dear ...!");
}else if (user >= 17 && user < 21){
alert("Good Evening Dear ...!");
}else {
alert("Good Night Dear ...!");
}
